函数的arguments为什么不是数组？如何转化为数组

看法是豆腐 3/11/2021 JS数组

因为arguments是一个对象，只不过它的属性从0开始排，一次为0，1，2...最后还有calleee和length属性。这类对象成为类数组

常见的类数组：

  • 用getElementByTagName/ClassName/Name() 获得的HTMLCollection
  • 用querySelector获得的nodeList

那这会导致很多数组的方法就不能用了，必要时需要我们将他们转换成数组，有哪些方法呢？

# 1. Array.prototype.slice.call()

function sum(a, b) {
  let args = Array.prototype.slice.call(arguments);
  console.log(args.reduce((sum, cur) => sum + cur));//args可以调用数组原生的方法啦
}
sum(1, 2);//3

# 2. Array.from()

function sum(a, b) {
  let args = Array.from(arguments);
  console.log(args.reduce((sum, cur) => sum + cur));//args可以调用数组原生的方法啦

# 3. ES6的展开运算符

function sum(a, b) {
  let args = [...arguments];
  console.log(args.reduce((sum, cur) => sum + cur));//args可以调用数组原生的方法啦

function sum(a, b) {
  let args = Array.prototype.concat.apply([], arguments);//apply方法会把第二个参数展开
